### State Management (LangGraph State)

```python
from typing import TypedDict, List, Dict, Any, Optional
from langgraph.graph import StateGraph

class AgentState(TypedDict):
    """State shared across agent nodes."""
    
    # Input data
    resume: Dict[str, Any]           # Structured resume data
    jd: Dict[str, Any]               # Structured JD data
    user_id: str                     # User UUID
    analysis_id: Optional[str]       # Analysis UUID (created in workflow)
    
    # Research data
    company_info: Optional[Dict]     # Company research results
    job_market_data: Optional[Dict]  # Market trends, salary data
    
    # Analysis results
    scores: Optional[Dict]           # Category scores
    overall_score: Optional[float]   # Overall fit score
    fit_classification: Optional[str] # NOT_FIT, GOOD_FIT, GREAT_FIT
    match_details: Optional[Dict]    # Detailed match breakdown
    
    # Generated content
    fit_rationale: Optional[str]     # LLM-generated rationale
    recommendations: Optional[List[str]]  # Resume recommendations
    learning_resources: Optional[List[Dict]]  # Learning resources
    
    # Workflow control
    skip_research: bool = False      # Flag to skip research node
    error: Optional[str] = None      # Error message if workflow fails
    
    # Metadata
    progress: int = 0                # Progress percentage (0-100)
    current_stage: str = "starting"  # Current workflow stage
```

```python
from fastapi import BackgroundTasks

async def analyze_async(
    resume_id: str,

    analysis_id = create_analysis_record()

    return {"analysis_id": analysis_id, "status": "queued"}
```
